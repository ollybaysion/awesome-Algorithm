# 2-SAT

## 개념

각 변수가 **참 또는 거짓** 두 값 중 하나를 가질 때, **2-CNF(각 절이 2개 이하 리터럴)**의 충족 가능성을 O(V+E)에 결정.

---

## 함의 그래프 (Implication Graph)

절 `(A ∨ B)` → 함의 `¬A → B`와 `¬B → A` 추가.

```
변수 x:   노드 x     (참)
         노드 x+n   (거짓)
```

**판단 기준**: x와 ¬x가 **같은 SCC**에 있으면 **UNSAT**.

**해 구성**: SCC 위상 정렬에서 x의 SCC 번호 > ¬x의 SCC 번호이면 x = 참.

---

## 구현

```cpp
const int MAXN = 200005;  // 변수 2N개 노드

struct SAT {
    // 함의 그래프 (에지리스트)
    struct Edge { int to, next; } edges[MAXN * 4];
    int head[MAXN], ecnt = 0;
    int n;  // 변수 수

    // Tarjan SCC
    int disc[MAXN], low[MAXN], timer_ = 0;
    bool on_stk[MAXN];
    int  stk[MAXN], stk_top = 0;
    int  scc_id[MAXN], scc_cnt = 0;

    void init(int _n) {
        n = _n;
        memset(head, -1, sizeof(int) * (2*n + 2));
        ecnt = timer_ = stk_top = scc_cnt = 0;
        memset(disc, 0, sizeof(int) * (2*n + 2));
        memset(on_stk, 0, sizeof(bool) * (2*n + 2));
    }

    // 노드 인코딩: 변수 i의 참 = i, 거짓 = i+n (1-based)
    void add_edge(int u, int v) {
        edges[ecnt] = {v, head[u]}; head[u] = ecnt++;
    }

    // (a 또는 b) 절 추가
    // a > 0이면 변수 |a| = 참, a < 0이면 |a| = 거짓
    void add_clause(int a, int b) {
        int pa = a > 0 ? a : -a + n;
        int na = a > 0 ? a + n : -a;
        int pb = b > 0 ? b : -b + n;
        int nb = b > 0 ? b + n : -b;
        add_edge(na, pb);  // ¬a → b
        add_edge(nb, pa);  // ¬b → a
    }

    // 단일 리터럴 강제: a = 참
    void force_true(int a) {
        int pa = a > 0 ? a : -a + n;
        int na = a > 0 ? a + n : -a;
        add_edge(na, pa);  // ¬a → a
    }

    void dfs(int u) {
        disc[u] = low[u] = ++timer_;
        stk[stk_top++] = u; on_stk[u] = true;
        for (int e = head[u]; ~e; e = edges[e].next) {
            int v = edges[e].to;
            if (!disc[v]) { dfs(v); low[u] = min(low[u], low[v]); }
            else if (on_stk[v]) low[u] = min(low[u], disc[v]);
        }
        if (low[u] == disc[u]) {
            scc_cnt++;
            while (true) {
                int v = stk[--stk_top];
                on_stk[v] = false;
                scc_id[v] = scc_cnt;
                if (v == u) break;
            }
        }
    }

    // 충족 가능성 판단 + 해 구성
    // 반환: 충족 가능하면 true, ans[i] = i번 변수의 값 (1-based)
    bool solve(bool* ans) {
        for (int i = 1; i <= 2*n; i++)
            if (!disc[i]) dfs(i);
        for (int i = 1; i <= n; i++) {
            if (scc_id[i] == scc_id[i+n]) return false;
            ans[i] = scc_id[i] > scc_id[i+n];
        }
        return true;
    }
} sat;
```

## 사용 예시

```cpp
// N개 변수, M개 절
// 절: (x1 OR x2), (NOT x2 OR x3), ...
int main() {
    int n, m;
    scanf("%d %d", &n, &m);
    sat.init(n);
    for (int i = 0; i < m; i++) {
        int a, b;
        scanf("%d %d", &a, &b);
        sat.add_clause(a, b);
    }
    bool ans[n+1];
    if (!sat.solve(ans)) printf("0\n");
    else {
        printf("1\n");
        for (int i = 1; i <= n; i++) printf("%d ", ans[i]);
    }
}
```
